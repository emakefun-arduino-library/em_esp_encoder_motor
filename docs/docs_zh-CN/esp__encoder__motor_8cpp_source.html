159void EspEncoderMotor::UpdateRpm() {
160 std::unique_lock lock(mutex_);
161 last_update_speed_time_ = std::chrono::system_clock::now();
162 while (!condition_.wait_until(
163 lock, last_update_speed_time_ + std::chrono::milliseconds(50), [this]() { return update_rpm_thread_ == nullptr; })) {
164 const auto now = std::chrono::system_clock::now();
165 const double duration = std::chrono::duration_cast<std::chrono::milliseconds>(now - last_update_speed_time_).count();
166 const double pulse_count = pulse_count_;
167 speed_rpm_ = (pulse_count - previous_pulse_count_) * 60000.0 / duration / total_ppr_;
168 previous_pulse_count_ = pulse_count;
169 last_update_speed_time_ = now;
170 if (driving_thread_ != nullptr) {
171 drive_ = true;
172 condition_.notify_all();
173 }
174 }
175}

177void EspEncoderMotor::Driving() {
178 std::unique_lock lock(mutex_);
179
180 while (driving_thread_ != nullptr) {
181 condition_.wait(lock, [this]() { return driving_thread_ == nullptr || drive_; });
182
183 if (driving_thread_ == nullptr) {
184 return;
185 }
186
187 if (target_speed_rpm_ < kDeadRpmZone && target_speed_rpm_ > -kDeadRpmZone) {
188 motor_driver_.PwmDuty(0);
189 } else {
190 const float speed_error = target_speed_rpm_ - speed_rpm_;
191 rpm_pid_.integral = constrain(rpm_pid_.integral + speed_error, -rpm_pid_.max_integral, rpm_pid_.max_integral);
192 const int16_t duty = round(rpm_pid_.p * speed_error + rpm_pid_.i * rpm_pid_.integral);
193 motor_driver_.PwmDuty(duty);
194 }
195 drive_ = false;
196 }
197}

199void EspEncoderMotor::DeleteThread(std::thread*& thread) {
200 if (thread != nullptr) {
201 const auto temp = std::exchange(thread, nullptr);
202 condition_.notify_all();
203 mutex_.unlock();
204 temp->join();
205 delete temp;
206 mutex_.lock();
207 }
